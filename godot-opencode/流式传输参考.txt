@tool
class_name DeepSeekChatStream
extends Node

## 用于向deepseek发送请求并获取流式返回的节点

## deepseek Token，在开放平台获取
@export var ds_token: String = ''
## 系统角色prompt，用于预设人设
@export_multiline var prompt: String = ""
## 深度思考
@export var use_thinking: bool = false
## 温度值，越高输出越随机，默认为1
@export_range(0.0, 2.0, 0.1) var temperature: float = 1.0
## 为正数时降低模型重复相同内容的可能性
@export_range(-2.0, 2.0, 0.1) var frequency_penalty: float = 0
## 为正数时增加模型谈论新主题的可能性
@export_range(-2.0, 2.0, 0.1) var presence_penalty: float = 0
## 最大输出长度，deepseek-chat模型，最大8K，deepseek-reasoner模型，最大64K
@export var max_tokens: int = 4096 * 8

## 开始返回正文
signal message_start()
## 返回正文
signal message(msg: String)
## 结束返回正文
signal message_end()
## 开始返回思考内容
signal think_start()
## 返回思考内容
signal think(msg: String)
## 结束返回思考内容
signal think_end()
## 返回结束
signal generate_finish

## 发送请求的http客户端
@onready var http_client: HTTPClient = HTTPClient.new()

var generatting: bool
var _stream_buffer: String
var _thinking_started: bool
var _message_started: bool

## 发送请求
func post_message(msg: String):
	if generatting:
		push_warning("正在生成中，请等待完成或调用close()中断")
		return

	var headers = [
		"Accept: application/json",
		"Authorization: Bearer %s" % ds_token,
		"Content-Type: application/json"
	]

	var body_dict = {
		"messages": [
			{"content": prompt, "role": "system"},
			{"content": msg, "role": "user"}
		],
		"model": "deepseek-v4-pro" if use_thinking else "deepseek-v4-flash",
		"frequency_penalty": frequency_penalty,
		"max_tokens": max_tokens,
		"presence_penalty": presence_penalty,
		"stream": true,
		"temperature": temperature,
		"top_p": 1,
	}

	var request_body = JSON.stringify(body_dict)
	_stream_buffer = ""
	_thinking_started = false
	_message_started = false

	var connect_err = http_client.connect_to_host("https://api.deepseek.com")
	generatting = true
	if connect_err != OK:
		push_error("连接服务器失败: " + error_string(connect_err))
		generatting = false
		return
	while http_client.get_status() == HTTPClient.STATUS_CONNECTING or http_client.get_status() == HTTPClient.STATUS_RESOLVING:
		http_client.poll()
		await get_tree().process_frame

	if http_client.get_status() != HTTPClient.STATUS_CONNECTED:
		push_error("连接服务器失败，状态: %d" % http_client.get_status())
		generatting = false
		return

	var err = http_client.request(HTTPClient.METHOD_POST, "/chat/completions", headers, request_body)
	if err != OK:
		push_error("请求发送失败: " + error_string(err))
		generatting = false
		return
	while http_client.get_status() == HTTPClient.STATUS_REQUESTING:
		http_client.poll()
		await get_tree().process_frame

	if not http_client.has_response():
		push_error("服务器未返回响应")
		generatting = false
		return

	var response_code = http_client.get_response_code()
	if response_code != 200:
		var error_body = PackedByteArray()
		while http_client.get_status() == HTTPClient.STATUS_BODY:
			http_client.poll()
			var chunk = http_client.read_response_body_chunk()
			if chunk.size() > 0:
				error_body.append_array(chunk)
			else:
				await get_tree().process_frame
		var error_msg = error_body.get_string_from_utf8() if error_body.size() > 0 else ""
		push_error("API请求失败 (HTTP %d): %s" % [response_code, error_msg])
		generatting = false
		generate_finish.emit()
		return

	headers = http_client.get_response_headers_as_dictionary()

	while http_client.get_status() == HTTPClient.STATUS_BODY:
		http_client.poll()
		var chunk = http_client.read_response_body_chunk()
		if chunk.size() == 0:
			await get_tree().process_frame
		else:
			_stream_buffer += chunk.get_string_from_utf8()
			_process_stream_buffer()
			if not generatting:
				break

	if generatting:
		generatting = false
		generate_finish.emit()

func _process_stream_buffer():
	while true:
		var newline_pos = _stream_buffer.find("\n")
		if newline_pos == -1:
			break

		var line = _stream_buffer.substr(0, newline_pos).strip_edges()
		_stream_buffer = _stream_buffer.substr(newline_pos + 1)

		if not line.begins_with("data: "):
			continue

		var data_string = line.substr(6)
		if data_string == "[DONE]":
			generatting = false
			if _thinking_started:
				think_end.emit()
			if _message_started:
				message_end.emit()
			generate_finish.emit()
			return

		var json = JSON.new()
		var parse_err = json.parse(data_string)
		if parse_err != OK:
			push_warning("JSON解析跳过: " + json.get_error_message())
			continue

		var data = json.get_data()
		if data == null or not data.has("choices"):
			continue

		var choices := data["choices"] as Array
		if choices.size() == 0:
			continue

		var delta = choices[0].get("delta")
		if delta == null:
			continue

		if use_thinking and delta.has("reasoning_content") and delta.get("reasoning_content") != null:
			if not _thinking_started:
				_thinking_started = true
				think_start.emit()
			think.emit(delta["reasoning_content"])
		elif delta.has("content") and delta.get("content") != null:
			if not _message_started:
				_message_started = true
				message_start.emit()
			message.emit(delta["content"])

		if choices[0].has("finish_reason") and choices[0].get("finish_reason") != null:
			generatting = false
			if _thinking_started:
				think_end.emit()
			if _message_started:
				message_end.emit()
			generate_finish.emit()
			return

## 中断请求
func close():
	generatting = false
	_stream_buffer = ""
	http_client.close()
